import os
import re

def update_index(directory):
    index_path = os.path.join(directory, "readme-computer-science-index.md")
    if os.path.exists(index_path):
        with open(index_path, "r") as f:
            content = f.read()
        # Extract current table (simplified)
        table_start = content.find("```dataview")
        table_end = content.find("```", table_start + 10)
        if table_start != -1 and table_end != -1:
            table_content = content[table_start:table_end + 3]
            new_entries = []
            for root, _, files in os.walk(os.path.join(directory, "science")):
                for file in files:
                    if file.endswith(".md") and "9" in file:
                        note_id = file.replace(".md", "")
                        new_entries.append(f"| {note_id} | {note_id.split(':')[-1].capitalize()} | #ai_fairness, #machine_learning | 2025-03-18 |")
            if new_entries:
                new_table = table_content.replace(
                    table_content[table_content.find("|") + 1:table_content.rfind("|")],
                    "\n".join(new_entries)
                )
                content = content.replace(table_content, new_table)
                with open(index_path, "w") as f:
                    f.write(content)
                print(f"Updated {index_path}")

if __name__ == "__main__":
    update_index("anacostia-vault")
```
