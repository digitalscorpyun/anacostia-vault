#### 3. Automation Script
Create `pomodoro_timer.py` in `anacostia-vault/projects`:
```python
import time
import os

def pomodoro_timer():
    work_time = 25 * 60  # 25 minutes in seconds
    break_time = 5 * 60   # 5 minutes in seconds
    log_path = os.path.join("anacostia-vault", "tools", "pomodor-log.md")

    while True:
        task = input("Enter task (e.g., 'Bias Detection Study'): ")
        print(f"Starting 25-minute Pomodoro for {task}...")
        time.sleep(work_time)
        print("Pomodoro complete! Take a 5-minute break.")
        time.sleep(break_time)
        print("Break over! Ready for next Pomodoro?")
        
        with open(log_path, "a") as f:
            date = time.strftime("%Y-%m-%d")
            f.write(f"| {date} | 1 | {task} | [[9/8]] |\n")
        print(f"Logged to {log_path}")

if __name__ == "__main__":
    pomodoro_timer()
```
- Run to start a 25-minute work cycle, followed by a 5-minute break, logging to `pomodor-log.md`.
