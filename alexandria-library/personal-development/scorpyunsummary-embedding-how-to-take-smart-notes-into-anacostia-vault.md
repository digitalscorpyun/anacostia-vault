#### 3. Automation Script

```python
import os
import re

def process_notes(directory):
    inbox_path = os.path.join(directory, "inbox.md")
    if os.path.exists(inbox_path):
        with open(inbox_path, "r") as f:
            content = f.read()
        ideas = re.findall(r"- (Idea: .+)", content)
        for idea in ideas:
            note_id = input(f"Assign ID for '{idea}': ")
            perm_note = f"{note_id}.md"
            with open(os.path.join(directory, perm_note), "w") as f:
                f.write(f"# {idea}\n- Linked to [[zettelkasten-method]]\n")
            content = content.replace(idea, f"- Processed into [[{note_id}]]")
        with open(inbox_path, "w") as f:
            f.write(content)

if __name__ == "__main__":
    process_notes("anacostia-vault")
```
- Run to convert inbox ideas into numbered permanent notes.
