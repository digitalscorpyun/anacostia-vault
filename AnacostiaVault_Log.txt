        [int]$Level = 0
    )

    # Define Indentation
    $Indent = " " * ($Level * 4)

    # Get all files and folders in the path
    Get-ChildItem -Path $Path | ForEach-Object {
        if ($_ -is [System.IO.DirectoryInfo]) {
            # Print directory
